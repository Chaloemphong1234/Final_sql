ผู้ดูแลระบบต้องการ แสดงรายชื่อนักศึกษา ที่ตรงตามเงื่อนไขทั้งหมดดังนี้
      1. นักศึกษาต้องมีสถานะเป็น active
      2. ต้องเคยลงทะเบียนเรียนในปี ระหว่าง 2021 ถึง 2024
      3. ต้องเรียนรายวิชาที่มีหน่วยกิต มากกว่า 3 หน่วยกิต
      4. ชื่อนักศึกษาต้องขึ้นต้นด้วยคำว่า “ชา”
      5. ต้องไม่เคยลงทะเบียนเรียนในรายวิชาที่ชื่อมีคำว่า “ทดลอง” ปรากฏอยู่
      6. ชื่อนักศึกษาที่แสดงต้องไม่ซ้ำกัน
      7. ต้องเรียงลำดับผลลัพธ์ตามชื่อนักศึกษาจาก ก → ฮ
คำสั่ง SQL ใด ถูกต้องและครบถ้วนตามเงื่อนไขมากที่สุด

ก.
SELECT DISTINCT student.student_name
FROM student, enrollment, course
WHERE student.student_id = enrollment.student_id
AND enrollment.course_id = course.course_id
AND student.student_status = 'active'
AND enrollment.enroll_year >= 2021
AND enrollment.enroll_year <= 2024
AND course.course_credit >= 3
AND student.student_name LIKE '%ชา'
AND course.course_name NOT LIKE '%ทดลอง%'
ORDER BY student.student_name ASC;

ข.
SELECT DISTINCT student.student_name
FROM student
JOIN enrollment ON student.student_id = enrollment.student_id
JOIN course ON enrollment.course_id = course.course_id
WHERE student.student_status = 'active'
AND enrollment.enroll_year BETWEEN 2021 AND 2024
AND course.course_credit > 3
AND student.student_name LIKE 'ชา%'
AND student.student_id NOT IN (
SELECT enrollment.student_id
FROM enrollment
JOIN course ON enrollment.course_id = course.course_id
WHERE course.course_name LIKE '%ทดลอง%'
)
ORDER BY student.student_name ASC;

ค.
SELECT student.student_name
FROM student
WHERE student.student_status = 'active'
AND student.student_name LIKE 'ชา%'
AND student.student_id IN (
SELECT enrollment.student_id
FROM enrollment
WHERE enrollment.enroll_year BETWEEN 2021 AND 2024
)
ORDER BY student.student_name ASC;

ง.
SELECT student.student_name
FROM student
LEFT JOIN enrollment ON student.student_id = enrollment.student_id
LEFT JOIN course ON enrollment.course_id = course.course_id
WHERE student.student_status = 'active'
AND enrollment.enroll_year BETWEEN 2021 AND 2024
AND course.course_credit > 3
AND student.student_name LIKE 'ชา%'
AND course.course_name NOT IN ('ทดลอง')
ORDER BY student.student_name ASC;
